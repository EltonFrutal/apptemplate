text — justificativa do cancelamento
💡
Regra de ouro — nunca excluir, sempre cancelar: registros de dados dos clientes nunca são deletados do banco. Quando cancelados, o campo ativo = false e cancelado_em é preenchido. As listagens filtram WHERE ativo = true e os cancelados ficam invisíveis na tela — mas preservados para auditoria e histórico.
Auditoria básica — todas as tabelas
Cancelamento — tabelas de dados dos clientes

db_apptemplate.sql · v1.2
-- ================================================
-- db_apptemplate · Schema Completo v1.2
-- AppTemplate · Sistema Multi-App Multi-Empresa
-- Inclui: auditoria tipo 1, codigo_consys, cancelamento
-- ================================================

-- 1. USUÁRIOS
CREATE TABLE usuarios (
  id            UUID      DEFAULT gen_random_uuid() PRIMARY KEY,
  email         TEXT      UNIQUE NOT NULL,
  nome          TEXT      NOT NULL,
  senha_hash    TEXT      NOT NULL,
  foto_url      TEXT,
  telefone      TEXT,
  is_admin      BOOLEAN   DEFAULT false,
  ativo         BOOLEAN   DEFAULT true,
  ultimo_acesso TIMESTAMP,
  criado_em     TIMESTAMP DEFAULT now(),
  criado_por    UUID      REFERENCES usuarios(id),
  alterado_em   TIMESTAMP,
  alterado_por  UUID      REFERENCES usuarios(id)
);

-- 2. APPS
CREATE TABLE apps (
  id         UUID    DEFAULT gen_random_uuid() PRIMARY KEY,
  codigo     TEXT    UNIQUE NOT NULL,
  nome       TEXT    NOT NULL,
  descricao  TEXT,
  icone_url  TEXT,
  url_base   TEXT,
  porta      INTEGER,
  banco_nome TEXT,
  ativo      BOOLEAN DEFAULT true,
  criado_em    TIMESTAMP DEFAULT now(),
  criado_por   UUID      REFERENCES usuarios(id),
  alterado_em  TIMESTAMP,
  alterado_por UUID      REFERENCES usuarios(id)
);

-- 3. EMPRESAS
CREATE TABLE empresas (
  id             UUID      DEFAULT gen_random_uuid() PRIMARY KEY,
  cnpj           TEXT      UNIQUE,
  codigo         TEXT      UNIQUE NOT NULL,
  codigo_consys  TEXT      UNIQUE,
  razao_social   TEXT      NOT NULL,
  nome_fantasia  TEXT,
  logo_url       TEXT,
  email          TEXT,
  telefone       TEXT,
  ativo          BOOLEAN   DEFAULT true,
  criado_em      TIMESTAMP DEFAULT now(),
  criado_por     UUID      REFERENCES usuarios(id),
  alterado_em    TIMESTAMP,
  alterado_por   UUID      REFERENCES usuarios(id)
);

-- 4. FILIAIS
CREATE TABLE filiais (
  id            UUID    DEFAULT gen_random_uuid() PRIMARY KEY,
  empresa_id    UUID    REFERENCES empresas(id),
  nome          TEXT    NOT NULL,
  cnpj          TEXT    UNIQUE,
  codigo        TEXT,
  codigo_consys TEXT    UNIQUE,
  endereco      TEXT,
  cidade        TEXT,
  estado        CHAR(2),
  is_matriz     BOOLEAN DEFAULT false,
  ativo         BOOLEAN DEFAULT true,
  criado_em     TIMESTAMP DEFAULT now(),
  criado_por    UUID      REFERENCES usuarios(id),
  alterado_em   TIMESTAMP,
  alterado_por  UUID      REFERENCES usuarios(id)
);

-- 5. MÓDULOS
CREATE TABLE modulos (
  id       UUID    DEFAULT gen_random_uuid() PRIMARY KEY,
  app_id   UUID    REFERENCES apps(id),
  codigo   TEXT    UNIQUE NOT NULL,
  nome     TEXT    NOT NULL,
  icone    TEXT,
  url_rota TEXT,
  ordem    INTEGER DEFAULT 0,
  ativo    BOOLEAN DEFAULT true,
  criado_em     TIMESTAMP DEFAULT now(),
  criado_por    UUID      REFERENCES usuarios(id),
  alterado_em   TIMESTAMP,
  alterado_por  UUID      REFERENCES usuarios(id)
);

-- 6. PERMISSÕES
CREATE TABLE permissoes (
  id            UUID    DEFAULT gen_random_uuid() PRIMARY KEY,
  usuario_id    UUID    REFERENCES usuarios(id),
  app_id        UUID    REFERENCES apps(id),
  empresa_id    UUID    REFERENCES empresas(id),
  filial_id     UUID    REFERENCES filiais(id),  -- NULL = todas
  modulo_id     UUID    REFERENCES modulos(id),  -- NULL = todos
  perfil        TEXT    DEFAULT 'view',
  pode_incluir  BOOLEAN DEFAULT false,
  pode_alterar  BOOLEAN DEFAULT false,
  pode_excluir  BOOLEAN DEFAULT false,
  pode_exportar BOOLEAN DEFAULT false,
  criado_em     TIMESTAMP DEFAULT now(),
  criado_por    UUID      REFERENCES usuarios(id),
  alterado_em   TIMESTAMP,
  alterado_por  UUID      REFERENCES usuarios(id),
  UNIQUE(usuario_id, app_id, empresa_id, filial_id, modulo_id)
);

-- 7. CLIENTES (exemplo tabela de dados com cancelamento)
--    Padrão a ser seguido por todas as tabelas de negócio
CREATE TABLE clientes (
  id         UUID    DEFAULT gen_random_uuid() PRIMARY KEY,
  empresa_id UUID    REFERENCES empresas(id),
  filial_id  UUID    REFERENCES filiais(id),
  nome       TEXT    NOT NULL,
  email      TEXT,
  telefone   TEXT,
  cidade     TEXT,
  ativo      BOOLEAN DEFAULT true,
  criado_em              TIMESTAMP DEFAULT now(),
  criado_por             UUID      REFERENCES usuarios(id),
  alterado_em            TIMESTAMP,
  alterado_por           UUID      REFERENCES usuarios(id),
  cancelado_em           TIMESTAMP,
  cancelado_por          UUID      REFERENCES usuarios(id),
  motivo_cancelamento    TEXT
);
